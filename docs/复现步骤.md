### 2.7 数据验证

创建验证脚本 `validate_data.py`：

```python
# validate_data.py
import os
import sys
import stable_worldmodel as swm
import torch
import h5py

def validate_h5_file(filepath):
    """验证 HDF5 文件的完整性"""
    print(f"验证文件: {filepath}")
    
    if not os.path.exists(filepath):
        print(f"  ✗ 文件不存在")
        return False
    
    try:
        with h5py.File(filepath, 'r') as f:
            print(f"  ✓ 文件可打开")
            print(f"  数据集列: {list(f.keys())}")
            print(f"  样本数: {len(f[list(f.keys())[0]])}")
        return True
    except Exception as e:
        print(f"  ✗ 错误: {e}")
        return False

def validate_dataset(dataset_name, cache_dir):
    """验证数据集加载"""
    print(f"\n验证数据集: {dataset_name}")
    
    try:
        dataset = swm.data.HDF5Dataset(
            dataset_name,
            keys_to_cache=['action', 'proprio', 'state'],
            cache_dir=cache_dir,
        )
        
        print(f"  ✓ 数据集加载成功")
        print(f"  列名: {dataset.column_names}")
        print(f"  长度: {len(dataset)}")
        
        # 采样一条数据
        sample = dataset[0]
        print(f"  采样一条数据:")
        for k, v in sample.items():
            if hasattr(v, 'shape'):
                print(f"    {k}: shape={v.shape}, dtype={v.dtype}")
        
        return True
    except Exception as e:
        print(f"  ✗ 错误: {e}")
        return False

if __name__ == "__main__":
    cache_dir = os.environ.get("STABLEWM_HOME", os.path.expanduser("~/.stable-wm"))
    
    # 验证 HDF5 文件
    h5_file = os.path.join(cache_dir, "pusht_expert_train.h5")
    if not validate_h5_file(h5_file):
        sys.exit(1)
    
    # 验证数据集加载
    if not validate_dataset("pusht_expert_train", cache_dir):
        sys.exit(1)
    
    print("\n✓ 所有验证通过")
```

运行验证：


### 3.5 常见问题处理

| 问题 | 命令 | 说明 |
|------|------|------|
| 显存溢出 | 减小 `batch_size` | 尝试 16 或 24 |
| 数据加载慢 | 增加 `num_workers` | 尝试 4-6 |
| 检查点未保存 | 检查权限 | `ls -ld $STABLEWM_HOME` |
| NaN 损失 | 降低学习率 | `optimizer.lr=1e-5` |

### 4.5 检查点管理

#### 自动保存

检查点自动保存到：

```

├── weights_epoch_1.pt
├── weights_epoch_10.pt
├── weights_epoch_50.pt
├── weights_epoch_100.pt
```

#### 手动转换检查点

```python
# convert_ckpt.py
import torch
import stable_worldmodel as swm
from pathlib import Path

# 查找最新的检查点
ckpt_dir = Path("$STABLEWM_HOME/checkpoints")
latest_run = max([d for d in ckpt_dir.iterdir() if d.is_dir()], key=lambda p: p.stat().st_mtime)
latest_ckpt = latest_run / "weights_epoch_100.pt"

print(f"加载检查点: {latest_ckpt}")

# 加载模型
model = torch.load(latest_ckpt, map_location='cpu', weights_only=False)

# 保存为 object.ckpt 格式
out_dir = Path("$STABLEWM_HOME/pusht")
out_dir.mkdir(parents=True, exist_ok=True)
torch.save(model, out_dir / 'lewm_object.ckpt')

print(f"检查点已保存到: {out_dir / 'lewm_object.ckpt'}")
```

```bash
python convert_ckpt.py
```


### 5.4 评估结果分析

创建分析脚本 `analyze_results.py`：

```python
# analyze_results.py
import re
from pathlib import Path

result_file = Path("$STABLEWM_HOME/pusht/pusht_results.txt")

if not result_file.exists():
    print(f"结果文件不存在: {result_file}")
    exit(1)

with open(result_file, 'r') as f:
    content = f.read()

# 提取指标
metrics = {}
for line in content.split('\n'):
    if 'metrics:' in line:
        match = re.search(r'\{(.+)\}', line)
        if match:
            metrics_str = match.group(1)
            for item in metrics_str.split(','):
                key, value = item.split(':')
                metrics[key.strip()] = float(value.strip())

print("评估结果:")
for key, value in metrics.items():
    print(f"  {key}: {value:.4f}")
```

```bash
python analyze_results.py
```

## 六、结果分析

### 6.1 训练曲线分析

创建分析脚本 `plot_training.py`：

```python
# plot_training.py
import matplotlib.pyplot as plt
import numpy as np

# 如果使用 WandB，可以从 API 获取数据
# 这里假设数据从日志文件读取

# 示例数据
epochs = np.arange(1, 101)
total_loss = np.exp(-0.05 * epochs) + 0.1 + np.random.randn(100) * 0.02
pred_loss = np.exp(-0.05 * epochs) + 0.05 + np.random.randn(100) * 0.01
sigreg_loss = np.full(100, 0.09) + np.random.randn(100) * 0.005

fig, axes = plt.subplots(1, 3, figsize=(15, 4))

axes[0].plot(epochs, total_loss)
axes[0].set_title('Total Loss')
axes[0].set_xlabel('Epoch')
axes[0].set_ylabel('Loss')
axes[0].grid(True)

axes[1].plot(epochs, pred_loss)
axes[1].set_title('Prediction Loss')
axes[1].set_xlabel('Epoch')
axes[1].set_ylabel('Loss')
axes[1].grid(True)

axes[2].plot(epochs, sigreg_loss)
axes[2].set_title('SIGReg Loss')
axes[2].set_xlabel('Epoch')
```

## 七、conda 使用指南

### 7.1 环境管理

#### 创建环境

```bash
# 创建新环境
conda create -n myenv python=3.10 -y

# 指定版本
conda create -n myenv python=3.9 numpy=1.24.0 -y

# 克隆环境
conda create --clone myenv --name myenv_copy
```

#### 激活/退出环境

```bash
# 激活环境
conda activate myenv

# 退出环境
conda deactivate
```

#### 列出环境

```bash
# 列出所有环境
conda env list

# 查看当前环境
conda info --envs
```

#### 删除环境

```bash
# 删除环境
conda env remove -n myenv

# 删除指定路径的环境
conda env remove -p /path/to/env
```

### 7.2 包管理

#### 安装包

```bash
# 从 conda 安装（优先）
conda install numpy

# 指定版本
conda install numpy=1.24.0

# 从 channel 安装
conda install -c conda-forge package

# 从 PyPI 安装（conda 没有时）
pip install package_name

# 安装本地包
pip install -e .
```

#### 卸载/更新包

```bash
# 卸载包
conda remove numpy

# 更新包
conda update numpy

# 更新所有包
conda upgrade --all
```

#### 查看已安装包

```bash
# 列出当前环境的包
conda list

# 搜索包
conda search numpy

# 查看包信息
conda info numpy
```

### 7.3 导出/导入环境

#### 导出环境

```bash
# 导出为 YAML（推荐，可重建环境）
conda env export > environment.yml

# 仅导出包列表
conda list --explicit > requirements.txt
```

#### 导入环境

```bash
# 从 YAML 创建环境
conda env create -f environment.yml

# 从包列表创建
conda create --name myenv --file requirements.txt
```

### 7.4 Channel 管理

```bash
# 查看 channels
conda config --show channels

# 添加 channel
conda config --add channels conda-forge
conda config --add channels pytorch

# 设置优先级
conda config --set channel_priority strict

# 移除 channel
conda config --remove channels conda-forge
```

### 7.5 使用国内镜像（可选）

```bash
# 清华 TUNA 镜像
conda config --add channels https://mirrors.tuna.tsinghua.edu.cn/anaconda/pkgs/main/
conda config --add channels https://mirrors.tuna.tsinghua.edu.cn/anaconda/pkgs/free/
conda config --set show_channel_urls yes
```

### 7.6 清理缓存

```bash
# 清理所有缓存
conda clean --all

# 清理包缓存
conda clean --packages

# 清理索引缓存
conda clean --index-cache
```
