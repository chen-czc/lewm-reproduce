### 2.7 数据验证

创建验证脚本 `validate_data.py`：

```python
# validate_data.py
import os
import sys
import stable_worldmodel as swm
import torch
import h5py

def validate_h5_file(filepath):
    """验证 HDF5 文件的完整性"""
    print(f"验证文件: {filepath}")
    
    if not os.path.exists(filepath):
        print(f"  ✗ 文件不存在")
        return False
    
    try:
        with h5py.File(filepath, 'r') as f:
            print(f"  ✓ 文件可打开")
            print(f"  数据集列: {list(f.keys())}")
            print(f"  样本数: {len(f[list(f.keys())[0]])}")
        return True
    except Exception as e:
        print(f"  ✗ 错误: {e}")
        return False

def validate_dataset(dataset_name, cache_dir):
    """验证数据集加载"""
    print(f"\n验证数据集: {dataset_name}")
    
    try:
        dataset = swm.data.HDF5Dataset(
            dataset_name,
            keys_to_cache=['action', 'proprio', 'state'],
            cache_dir=cache_dir,
        )
        
        print(f"  ✓ 数据集加载成功")
        print(f"  列名: {dataset.column_names}")
        print(f"  长度: {len(dataset)}")
        
        # 采样一条数据
        sample = dataset[0]
        print(f"  采样一条数据:")
        for k, v in sample.items():
            if hasattr(v, 'shape'):
                print(f"    {k}: shape={v.shape}, dtype={v.dtype}")
        
        return True
    except Exception as e:
        print(f"  ✗ 错误: {e}")
        return False

if __name__ == "__main__":
    cache_dir = os.environ.get("STABLEWM_HOME", os.path.expanduser("~/.stable-wm"))
    
    # 验证 HDF5 文件
    h5_file = os.path.join(cache_dir, "pusht_expert_train.h5")
    if not validate_h5_file(h5_file):
        sys.exit(1)
    
    # 验证数据集加载
    if not validate_dataset("pusht_expert_train", cache_dir):
        sys.exit(1)
    
    print("\n✓ 所有验证通过")
```

### 3.2 转换权重格式

HuggingFace 下载的权重需要转换为 eval.py 期望的格式。

使用 test 目录下的转换脚本：

```bash

```

转换脚本会自动：
- 读取 `data/hf_tworoom/config.json` 和 `data/hf_tworoom/weights.pt`
- 创建模型结构并加载权重
- 保存为 `data/tworoom/lewm_object.ckpt`

如需转换其他任务，修改脚本顶部的配置宏：

```python

```

如果需要手动创建转换脚本：

```python
# convert_pretrained.py
import json
import torch
from pathlib import Path

def convert_hf_to_ckpt(src_dir, task_name, output_dir=None):
    """
    转换 HuggingFace 格式的权重为 object.ckpt 格式

    Args:
        src_dir: HuggingFace 下载目录（包含 config.json 和 weights.pt）
        task_name: 任务名称（pusht, cube, tworoom, reacher）
        output_dir: 输出目录（默认为 $STABLEWM_HOME）
    """
    import os
    import stable_worldmodel as swm

    if output_dir is None:
        output_dir = Path(swm.data.utils.get_cache_dir())

    src = Path(src_dir)
    out_dir = Path(output_dir) / task_name
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / 'lewm_object.ckpt'

    # 加载配置和权重
    cfg = json.loads((src / 'config.json').read_text())
    weights = torch.load(src / 'weights.pt', map_location='cpu', weights_only=False)

    print(f"任务: {task_name}")
    print(f"编码器: {cfg['encoder']['size']}")
    print(f"嵌入维度: {cfg.get('action_encoder', {}).get('emb_dim', 192)}")

    # 创建模型结构
    from jepa import JEPA
    from module import ARPredictor, Embedder, MLP
    import stable_pretraining as spt

    encoder = spt.backbone.utils.vit_hf(
        cfg["encoder"]["size"],
        patch_size=cfg["encoder"]["patch_size"],
        image_size=cfg["encoder"]["image_size"],
        pretrained=False,
        use_mask_token=False,
    )

    mlp_factory = lambda k: MLP(
        input_dim=cfg[k]["input_dim"],
        output_dim=cfg[k]["output_dim"],
        hidden_dim=cfg[k]["hidden_dim"],
        norm_fn=torch.nn.BatchNorm1d
    )

    model = JEPA(
        encoder=encoder,
        predictor=ARPredictor(**cfg["predictor"]),
        action_encoder=Embedder(**cfg["action_encoder"]),
        projector=mlp_factory("projector"),
        pred_proj=mlp_factory("pred_proj"),
    )

    # 加载权重
    model.load_state_dict(weights, strict=True)
    model.eval()

    # 保存为 object.ckpt 格式
    torch.save(model, out_path)
    print(f"✓ 检查点已保存到: {out_path}")

    return out_path

```

### 3.6 推理可视化（可选）

创建可视化脚本 `visualize_inference.py`：

```python
# visualize_inference.py
import matplotlib.pyplot as plt
from pathlib import Path

result_file = Path("$STABLEWM_HOME/tworoom/tworoom_pretrained_quick_test.txt")

if result_file.exists():
    with open(result_file, 'r') as f:
        content = f.read()

    # 提取指标
    import re
    metrics = {}
    for line in content.split('\n'):
        if 'metrics:' in line:
            match = re.search(r'\{(.+)\}', line)
            if match:
                metrics_str = match.group(1)
                for item in metrics_str.split(','):
                    key, value = item.split(':')
                    metrics[key.strip()] = float(value.strip())

    # 简单可视化
    fig, ax = plt.subplots(figsize=(8, 4))
    bars = ax.bar(metrics.keys(), metrics.values())
    ax.set_ylabel('Value')
    ax.set_title('Pretrained Model Performance')
    ax.bar_label(bars)
    plt.tight_layout()
    plt.savefig('pretrained_performance.png', dpi=150)
    print("✓ 可视化已保存到: pretrained_performance.png")

    # 打印结果
    print("\n预训练模型性能:")
    for key, value in metrics.items():
        print(f"  {key}: {value:.4f}")
else:
    print(f"结果文件不存在: {result_file}")
```

```python
# analyze_results.py
import re
from pathlib import Path

result_file = Path("$STABLEWM_HOME/tworoom/tworoom_results.txt")

if not result_file.exists():
    print(f"结果文件不存在: {result_file}")
    exit(1)

with open(result_file, 'r') as f:
    content = f.read()

# 提取指标
metrics = {}
for line in content.split('\n'):
    if 'metrics:' in line:
        match = re.search(r'\{(.+)\}', line)
        if match:
            metrics_str = match.group(1)
            for item in metrics_str.split(','):
                key, value = item.split(':')
                metrics[key.strip()] = float(value.strip())

print("评估结果:")
for key, value in metrics.items():
    print(f"  {key}: {value:.4f}")
```
